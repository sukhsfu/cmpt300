After making the executable file, use the following command to get the same functionality as ls:

ls -> ./UnixLs ls
ls -i -> ./UnixLs ls -i
ls -l -> ./UnixLs ls -l
ls -il -> ./UnixLs ls -il
ls -li -> ./UnixLs ls -li

ls <> <> <> -> ./UnixLs ls <> <> <>
ls -i <> <> <> -> ./UnixLs ls -i <> <> <>
ls -l <> <> <> -> ./UnixLs ls -l <> <> <>
ls -il <> <> <> -> ./UnixLs ls -il <> <> <>
ls -li <> <> <> -> ./UnixLs ls -li <> <> <>
